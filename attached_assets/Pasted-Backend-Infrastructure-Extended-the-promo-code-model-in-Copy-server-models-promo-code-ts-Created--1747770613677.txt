// In PromotionReview.tsx
const [useCustomDates, setUseCustomDates] = useState(false);
const [startDate, setStartDate] = useState(new Date(promotion.start_date));
const [endDate, setEndDate] = useState(new Date(promotion.end_date));
const [duration, setDuration] = useState(7); // Default 7 days

// Calculate end date based on duration
const updateEndDate = (start: Date, durationDays: number) => {
  const end = new Date(start);
  end.setDate(start.getDate() + durationDays);
  setEndDate(end);
};

// Duration options
const durationOptions = [
  { value: 7, label: '1 Week' },
  { value: 14, label: '2 Weeks' },
  { value: 30, label: '1 Month' },
  { value: 60, label: '2 Months' },
  { value: 90, label: '3 Months' },
  { value: 0, label: 'Custom' }
];

return (
  // ... existing review fields
  <div className="mt-6 border-t pt-4">
    <h3 className="text-lg font-medium">Promotion Scheduling</h3>
    
    <div className="flex items-center mt-2">
      <Switch
        checked={useCustomDates}
        onCheckedChange={setUseCustomDates}
        id="custom-dates"
      />
      <Label htmlFor="custom-dates" className="ml-2">
        Override clinic-suggested dates
      </Label>
    </div>
    
    <div className={cn("mt-4 grid grid-cols-1 gap-4 md:grid-cols-2", 
      !useCustomDates && "opacity-50 pointer-events-none")}>
      <div>
        <Label htmlFor="start-date">Start Date</Label>
        <DatePicker
          id="start-date"
          selected={startDate}
          onChange={(date) => {
            setStartDate(date);
            if (duration > 0) updateEndDate(date, duration);
          }}
          minDate={new Date()}
        />
      </div>
      
      <div>
        <Label htmlFor="duration">Duration</Label>
        <Select
          value={duration.toString()}
          onValueChange={(value) => {
            const days = parseInt(value);
            setDuration(days);
            if (days > 0) updateEndDate(startDate, days);
          }}
        >
          {durationOptions.map(option => (
            <SelectItem key={option.value} value={option.value.toString()}>
              {option.label}
            </SelectItem>
          ))}
        </Select>
      </div>
      
      <div>
        <Label htmlFor="end-date">End Date</Label>
        <DatePicker
          id="end-date"
          selected={endDate}
          onChange={setEndDate}
          minDate={new Date(startDate.getTime() + 86400000)} // Min 1 day after start

3. FeaturedPromotions Component for Homepage

Copy
// In FeaturedPromotions.tsx
import { useQuery } from 'react-query';
import { Promotion } from '../../types/promotion';
import { Card, CardContent, CardFooter } from '../ui/card';
import { Button } from '../ui/button';
import { formatCurrency, formatDate } from '../../utils/format';

export function FeaturedPromotions() {
  const { data: promotions, isLoading, error } = useQuery<Promotion[]>(
    'featuredPromotions',
    () => fetch('/api/promotions/featured').then(res => res.json())
  );

  if (isLoading) return <div className="py-12 text-center">Loading promotions...</div>;
  if (error) return <div className="py-12 text-center">Failed to load promotions</div>;
  if (!promotions?.length) return null;

  // Calculate days remaining for a promotion
  const getDaysRemaining = (endDate: string) => {
    const end = new Date(endDate);
    const today = new Date();
    const diffTime = end.getTime() - today.getTime();
    const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
    return diffDays > 0 ? diffDays : 0;
  };

  // Sort promotions by priority (highest first)
  const sortedPromotions = [...promotions].sort((a, b) => 
    b.homepage_priority - a.homepage_priority
  );

  return (
    <section className="py-12 bg-gray-50">
      <div className="container">
        <h2 className="text-3xl font-bold text-center mb-8">Featured Promotions</h2>
        
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
          {sortedPromotions.map(promotion => {
            const daysRemaining = getDaysRemaining(promotion.end_date);
            
            return (
              <Card key={promotion.id} className="overflow-hidden">
                {promotion.homepage_image_url && (
                  <div className="aspect-video relative overflow-hidden">
                    <img 
                      src={promotion.homepage_image_url} 
                      alt={promotion.title}
                      className="object-cover w-full h-full"
                    />
                    {daysRemaining <= 7 && daysRemaining > 0 && (
                      <div className="absolute top-2 right-2 bg-red-600 text-white px-3 py-1 rounded-full text-sm font-medium">
                        Ends in {daysRemaining} days
                      </div>
                    )}
                  </div>
                )}
                
                <CardContent className="pt-6">
                  <div className="flex items-center mb-3">
                    <img 
                      src={`/api/clinics/${promotion.clinic_id}/logo`} 
                      alt="Clinic logo"
                      className="w-8 h-8 rounded-full mr-2"
                    />
                    <span className="text-sm font-medium text-gray-500">
                      {promotion.clinic_name}
                    </span>
                  </div>
                  
                  <h3 className="text-xl font-bold mb-2">{promotion.title}</h3>
                  
                  <p className="text-gray-600 mb-4">
                    {promotion.homepage_short_description || promotion.description.substring(0, 100) + '...'}
                  </p>
                  
                  <div className="text-lg font-bold text-primary">
                    {promotion.discount_type === 'percentage' 
                      ? `${promotion.discount_value}% Off` 
                      : `Save ${formatCurrency(promotion.discount_value)}`}
                  </div>
                  
                  <div className="text-sm text-gray-500 mt-1">
                    Valid until {formatDate(promotion.end_date)}
                  </div>
                </CardContent>
                
                <CardFooter className="pt-0">
                  <Button variant="default" className="w-full">
                    View Details
                  </Button>
                </CardFooter>
              </Card>
            );
          })}
        </div>
      </div>
    </section>
  );
}
