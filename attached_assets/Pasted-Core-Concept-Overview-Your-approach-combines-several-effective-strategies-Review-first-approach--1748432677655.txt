Core Concept Overview
Your approach combines several effective strategies:

Review-first approach - Ensures quality control and collects valuable testimonials
Reward system - Incentivizes patients to refer others
Multi-channel distribution - Leverages social media for wider reach

// Prisma schema models

model Referral {
  id             String    @id @default(cuid())
  referralCode   String    @unique // Unique code per referral
  referrerId     String    // Patient who made the referral
  referrer       Patient   @relation("ReferrerRelation", fields: [referrerId], references: [id])
  referredEmail  String    // Email of person being referred
  referredName   String?   // Name of person being referred
  status         String    @default("PENDING") // PENDING, REGISTERED, CONVERTED
  message        String?   // Personal message
  rewardAmount   Float     // Amount to reward
  rewardClaimed  Boolean   @default(false)
  createdAt      DateTime  @default(now())
  updatedAt      DateTime  @updatedAt
}

model Review {
  id           String    @id @default(cuid())
  patientId    String
  patient      Patient   @relation(fields: [patientId], references: [id])
  rating       Int       // 1-5 stars
  title        String?
  content      String
  isPublic     Boolean   @default(false)
  createdAt    DateTime  @default(now())
  updatedAt    DateTime  @updatedAt
}

// Add to existing Patient model
model Patient {
  // existing fields
  reviews         Review[]
  referralsGiven  Referral[]  @relation("ReferrerRelation")
  referralCode    String?     @unique // Their personal referral code
  referralBalance Float       @default(0) // Total reward balance
}

// components/patient/ReviewForm.jsx
import React, { useState } from 'react';
import { useForm } from 'react-hook-form';
import axios from 'axios';
import toast from 'react-hot-toast';
import StarRating from '../ui/StarRating';

export default function ReviewForm({ onReviewSubmitted }) {
  const [rating, setRating] = useState(0);
  const [isSubmitting, setIsSubmitting] = useState(false);
  
  const { register, handleSubmit, formState: { errors } } = useForm({
    defaultValues: {
      title: '',
      content: '',
      isPublic: true
    }
  });
  
  const onSubmit = async (data) => {
    if (rating === 0) {
      toast.error('Please select a rating');
      return;
    }
    
    try {
      setIsSubmitting(true);
      
      await axios.post('/api/patient/reviews', {
        ...data,
        rating
      });
      
      toast.success('Thank you for your review!');
      
      if (onReviewSubmitted) {
        onReviewSubmitted();
      }
    } catch (error) {
      console.error('Error submitting review:', error);
      toast.error('Failed to submit your review. Please try again.');
    } finally {
      setIsSubmitting(false);
    }
  };
  
  return (
    <div className="bg-white p-6 rounded-lg shadow-md">
      <h2 className="text-xl font-bold mb-4">Share Your Experience</h2>
      <p className="text-gray-600 mb-6">
        Before referring others, please share your experience with us. Your feedback helps us improve and helps others make informed decisions.
      </p>
      
      <form onSubmit={handleSubmit(onSubmit)}>
        <div className="mb-4">
          <label className="block text-sm font-medium mb-2">Your Rating</label>
          <StarRating 
            rating={rating} 
            onChange={setRating} 
            size="large"
          />
          {rating === 0 && <p className="text-red-500 text-sm mt-1">Please select a rating</p>}
        </div>
        
        <div className="mb-4">
          <label className="block text-sm font-medium mb-2">Review Title</label>
          <input
            type="text"
            className="w-full p-2 border rounded"
            placeholder="Summarize your experience"
            {...register('title')}
          />
        </div>
        
        <div className="mb-4">
          <label className="block text-sm font-medium mb-2">Your Review</label>
          <textarea
            className="w-full p-2 border rounded"
            rows="4"
            placeholder="Tell us about your experience with our dental services..."
            {...register('content', { required: 'Please share your experience' })}
          ></textarea>
          {errors.content && <p className="text-red-500 text-sm mt-1">{errors.content.message}</p>}
        </div>
        
        <div className="mb-4">
          <label className="flex items-center">
            <input 
              type="checkbox" 
              className="mr-2"
              {...register('isPublic')}
            />
            <span className="text-sm text-gray-600">
              I allow my review to be shared publicly (your name will not be displayed)
            </span>
          </label>
        </div>
        
        <button
          type="submit"
          className="bg-blue-600 text-white px-4 py-2 rounded hover:bg-blue-700 disabled:bg-blue-300"
          disabled={isSubmitting}
        >
          {isSubmitting ? 'Submitting...' : 'Submit Review'}
        </button>
      </form>
    </div>
  );
}

import React from 'react';
import { FaTwitter, FaFacebook, FaWhatsapp, FaEnvelope, FaLink } from 'react-icons/fa';
import toast from 'react-hot-toast';

export default function SocialShare({ referralCode, patientName }) {
  const baseUrl = typeof window !== 'undefined' ? window.location.origin : '';
  const referralUrl = `${baseUrl}/register?ref=${referralCode}`;
  
  const shareText = `I had a great dental treatment experience in Turkey. Check it out and save on your dental care: ${referralUrl}`;
  
  const shareOnTwitter = () => {
    const url = `[https://twitter.com/intent/tweet?text=${encodeURIComponent(shareText)}`;](https://twitter.com/intent/tweet?text=${encodeURIComponent(shareText)}`;)
    window.open(url, '_blank');
  };
  
  const shareOnFacebook = () => {
    const url = `[https://www.facebook.com/sharer/sharer.php?u=${encodeURIComponent(referralUrl)}`;](https://www.facebook.com/sharer/sharer.php?u=${encodeURIComponent(referralUrl)}`;)
    window.open(url, '_blank');
  };
  
  const shareOnWhatsapp = () => {
    const url = `[https://wa.me/?text=${encodeURIComponent(shareText)}`;](https://wa.me/?text=${encodeURIComponent(shareText)}`;)
    window.open(url, '_blank');
  };
  
  const shareByEmail = () => {
    const subject = 'Quality Dental Treatment in Turkey';
    const body = `Hi,\n\nI wanted to share my experience with dental treatment in Turkey. They offer great quality at affordable prices.\n\nCheck it out here: ${referralUrl}\n\nBest,\n${patientName}`;

  };
  
  const copyLink = () => {
    navigator.clipboard.writeText(referralUrl);
    toast.success('Referral link copied to clipboard!');

// pages/api/patient/reviews.js
import { getServerSession } from "next-auth/next";
import { authOptions } from "../auth/[...nextauth]";
import { prisma } from "../../../lib/prisma";

export default async function handler(req, res) {
  const session = await getServerSession(req, res, authOptions);
  
  if (!session) {
    return res.status(401).json({ message: "You must be logged in." });
  }
  
  // POST: Create a new review
  if (req.method === 'POST') {
    try {
      const { rating, title, content, isPublic } = req.body;
      const patientId = session.user.id;
      
      // Validate input
      if (!rating || rating < 1 || rating > 5 || !content) {
        return res.status(400).json({ message: "Invalid review data" });
      }
      
      // Create the review
      const review = await prisma.review.create({
        data: {
          patientId,
          rating,
          title: title || "",
          content,
          isPublic: isPublic || false
        }
      });
      
      // Generate a referral code for this patient if they don't have one
      const patient = await prisma.patient.findUnique({
        where: { id: patientId }
      });
      
      if (!patient.referralCode) {
        const referralCode = generateReferralCode(patient.firstName, patient.lastName);
        
        await prisma.patient.update({
          where: { id: patientId },
          data: { referralCode }
        });
      }
      
      return res.status(201).json(review);
    } catch (error) {

// pages/api/patient/referrals.js
import { getServerSession } from "next-auth/next";
import { authOptions } from "../auth/[...nextauth]";
import { prisma } from "../../../lib/prisma";
import { sendReferralEmail } from "../../../lib/email";

export default async function handler(req, res) {
  const session = await getServerSession(req, res, authOptions);
  
  if (!session) {
    return res.status(401).json({ message: "You must be logged in." });
  }
  
  const patientId = session.user.id;
  
  // GET: Fetch patient's referrals
  if (req.method === 'GET') {
    try {
      const referrals = await prisma.referral.findMany({
        where: { referrerId: patientId },
        orderBy: { createdAt: 'desc' }
      });
      
      return res.status(200).json(referrals);
    } catch (error) {
      console.error("Error fetching referrals:", error);
      return res.status(500).json({ message: "Failed to fetch referrals" });
    }
  }
  
  // POST: Create a new referral
  if (req.method === 'POST') {
    try {
      const { name, email, message } = req.body;
      
      // Validate required fields
      if (!email) {
        return res.status(400).json({ message: "Email is required" });
      }
      
      // Check if patient has left a review (required before referring)
      const reviews = await prisma.review.findMany({
        where: { patientId }
      });
      
      if (reviews.length === 0) {
        return res.status(403).json({ 
          message: "You must leave a review before referring others" 
        });
      }
      
      // Get patient info including referral code
      const patient = await prisma.patient.findUnique({
        where: { id: patientId }
      });
      
      if (!patient.referralCode) {
        return res.status(500).json({ 
          message: "Referral system error. Please contact support." 
        });
      }
      
      // Check if this email has already been referred by this patient
      const existingReferral = await prisma.referral.findFirst({
        where: {
          referrerId: patientId,
          referredEmail: email
        }
      });
      
      if (existingReferral) {
        return res.status(400).json({ 
          message: "You have already referred this person" 
        });
      }
      
      // Create the referral
      const referral = await prisma.referral.create({
        data: {
          referralCode: `${patient.referralCode}-${Math.random().toString(36).substring(2, 7)}`,
          referrerId: patientId,
          referredName: name || "Friend",
          referredEmail: email,
          status: "PENDING",
          message: message || "",
          rewardAmount: 50 // £50 reward
        }
      });
      
      // Send email to the referred person
      await sendReferralEmail({
        to: email,
        name: name || "Friend",
        referrerName: `${patient.firstName} ${patient.lastName}`,
        referralCode: referral.referralCode,
        message: message || "",
        referralLink: `${process.env.NEXT_PUBLIC_APP_URL}/register?ref=${referral.referralCode}`
      });
      
      return res.status(201).json(referral);
    } catch (error) {
      console.error("Error creating referral:", error);
      return res.status(500).json({ message: "Failed to create referral" });
    }
  }
  
  return res.status(405).json({ message: "Method not allowed" });
}

// pages/api/auth/register.js
import { hash } from 'bcryptjs';
import { prisma } from '../../../lib/prisma';

export default async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ message: 'Method not allowed' });
  }
  
  try {
    const { firstName, lastName, email, password, referralCode } = req.body;
    
    // Validate required fields
    if (!firstName || !lastName || !email || !password) {
      return res.status(400).json({ message: 'All fields are required' });
    }
    
    // Check if email already exists
    const existingUser = await prisma.patient.findUnique({
      where: { email: email.toLowerCase() }
    });
    
    if (existingUser) {
      return res.status(400).json({ message: 'Email already in use' });
    }
    
    // Hash password
    const hashedPassword = await hash(password, 10);
    
    // Create patient
    const patient = await prisma.patient.create({
      data: {
        firstName,
        lastName,
        email: email.toLowerCase(),
        password: hashedPassword
      }
    });
    
    // Handle referral if code was provided
    if (referralCode) {
      const referral = await prisma.referral.findUnique({
        where: { referralCode }
      });
      
      if (referral) {
        // Update referral status
        await prisma.referral.update({
          where: { id: referral.id },
          data: { status: 'REGISTERED' }
        });
      }
    }
    
    return res.status(201).json({ 
      message: 'Registration successful',
      patient: {
        id: patient.id,
        firstName: patient.firstName,
        lastName: patient.lastName,
        email: patient.email
      }
    });
  } catch (error) {
    console.error('Error registering user:', error);
    return res.status(500).json({ message: 'Failed to register' });
  }
}

import { getServerSession } from "next-auth/next";
import { authOptions } from "../../auth/[...nextauth]";
import { prisma } from "../../../../lib/prisma";

export default async function handler(req, res) {
  const session = await getServerSession(req, res, authOptions);
  
  if (!session || !session.user.isAdmin) {
    return res.status(401).json({ message: "Unauthorized" });
  }
  
  if (req.method !== 'POST') {
    return res.status(405).json({ message: 'Method not allowed' });
  }
  
  try {
    const { patientId, treatmentIds, clinicId, totalAmount, dateScheduled } = req.body;
    
    // Create booking
    const booking = await prisma.booking.create({
      data: {
        patientId,
        clinicId,
        totalAmount,
        dateScheduled: new Date(dateScheduled),
        treatments: {
          connect: treatmentIds.map(id => ({ id }))
        },
        status: "CONFIRMED"
      }
    });
    
    // Check if this patient came from a referral
    const patient = await prisma.patient.findUnique({
      where: { id: patientId }
    });
    
    if (patient) {
      // Find referral by referral code
      const referral = await prisma.referral.findFirst({
        where: {
          referredEmail: patient.email,
          status: 'REGISTERED'
        }
      });
      
      if (referral) {
        // Update referral to CONVERTED and add reward to referrer
        await prisma.$transaction([
          // Update referral status
          prisma.referral.update({
            where: { id: referral.id },
            data: { status: 'CONVERTED' }
          }),
          
          // Add reward to referrer's balance
          prisma.patient.update({
            where: { id: referral.referrerId },
            data: {
              referralBalance: {
                increment: referral.rewardAmount
              }
            }
          })
        ]);
      }
    }
    
    return res.status(201).json(booking);
  } catch (error) {
    console.error('Error creating booking:', error);
    return res.status(500).json({ message: 'Failed to create booking' });
  }
}
