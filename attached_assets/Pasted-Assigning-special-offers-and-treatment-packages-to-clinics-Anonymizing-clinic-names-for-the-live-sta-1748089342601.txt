1. Clinic Anonymization Implementation
First, let's create a script to anonymize your clinic names:

// scripts/anonymize-clinics.js
const { PrismaClient } = require('@prisma/client');
const prisma = new PrismaClient();

async function anonymizeClinics() {
  try {
    const clinics = await prisma.clinic.findMany();
    console.log(`Found ${clinics.length} clinics to anonymize`);

    // Define naming components
    const locations = ['Istanbul', 'Antalya', 'Izmir', 'Bodrum', 'Kusadasi'];
    const qualities = ['Premium', 'Advanced', 'Elite', 'Modern', 'Professional'];
    const specialties = ['Dental', 'Smile', 'Aesthetic', 'Oral', 'Implant'];
    const suffixes = ['Clinic', 'Center', 'Practice', 'Care', 'Studio'];

    for (const clinic of clinics) {
      // Generate a unique name using components
      const location = locations[Math.floor(Math.random() * locations.length)];
      const quality = qualities[Math.floor(Math.random() * qualities.length)];
      const specialty = specialties[Math.floor(Math.random() * specialties.length)];
      const suffix = suffixes[Math.floor(Math.random() * suffixes.length)];
      
      const newName = `${location} ${quality} ${specialty} ${suffix}`;
      
      // Update clinic with anonymized name and contact details
      await prisma.clinic.update({
        where: { id: clinic.id },
        data: { 
          name: newName,
          // Anonymize contact info but keep location data
          email: `contact@${newName.toLowerCase().replace(/\s+/g, '')}.com`,
          phone: `+90${Math.floor(500000000 + Math.random() * 499999999)}`
        }
      });
      
      console.log(`Renamed clinic: ${clinic.name} → ${newName}`);
    }
    
    console.log('Clinic anonymization completed successfully');
  } catch (error) {
    console.error('Error anonymizing clinics:', error);
  } finally {
    await prisma.$disconnect();
  }
}

// prisma/schema.prisma

model SpecialOffer {
  id               String    @id @default(cuid())
  title            String
  description      String    @db.Text
  discountType     String    // "PERCENTAGE" or "FIXED_AMOUNT"
  discountValue    Float
  clinicId         String
  clinic           Clinic    @relation(fields: [clinicId], references: [id])
  treatmentIds     String[]  // Array of treatment IDs this offer applies to
  promoCode        String    @unique
  imageUrl         String?
  expiryDate       DateTime
  isActive         Boolean   @default(true)
  createdAt        DateTime  @default(now())
  updatedAt        DateTime  @updatedAt
}

model TreatmentPackage {
  id                String             @id @default(cuid())
  title             String
  description       String             @db.Text
  clinicId          String
  clinic            Clinic             @relation(fields: [clinicId], references: [id])
  promoCode         String             @unique
  imageUrl          String?
  totalPrice        Float
  originalPrice     Float              // Sum of individual treatment prices
  includedTreatments PackageTreatment[]
  includedServices   PackageService[]
  expiryDate        DateTime
  isActive          Boolean            @default(true)
  createdAt         DateTime           @default(now())
  updatedAt         DateTime           @updatedAt
}

model PackageTreatment {
  id               String           @id @default(cuid())
  packageId        String
  package          TreatmentPackage @relation(fields: [packageId], references: [id], onDelete: Cascade)
  treatmentId      String
  treatmentName    String           // Store name for display purposes
  treatmentPrice   Float            // Store original price
  quantity         Int              @default(1)
}

model PackageService {
  id               String           @id @default(cuid())
  packageId        String
  package          TreatmentPackage @relation(fields: [packageId], references: [id], onDelete: Cascade)
  serviceType      String           // "HOTEL", "TRANSFER", "CONSULTATION", etc.
  name             String
  description      String?
  quantity         Int              @default(1)
}

// Update existing Clinic model to include relationships
model Clinic {
  // existing fields...
  specialOffers     SpecialOffer[]
  treatmentPackages TreatmentPackage[]
}

// components/admin/SpecialOffersManager.jsx
import React, { useState, useEffect } from 'react';
import { useForm } from 'react-hook-form';
import axios from 'axios';
import toast from 'react-hot-toast';

export default function SpecialOffersManager() {
  const [clinics, setClinics] = useState([]);
  const [treatments, setTreatments] = useState([]);
  const [specialOffers, setSpecialOffers] = useState([]);
  const [isEditing, setIsEditing] = useState(false);
  const [selectedOfferId, setSelectedOfferId] = useState(null);
  const [isLoading, setIsLoading] = useState(false);

  const { register, handleSubmit, reset, setValue, watch, formState: { errors } } = useForm({
    defaultValues: {
      title: '',
      description: '',
      discountType: 'PERCENTAGE',
      discountValue: 0,
      clinicId: '',
      treatmentIds: [],
      promoCode: '',
      expiryDate: new Date().toISOString().split('T')[0],
      isActive: true
    }
  });

  const watchClinicId = watch('clinicId');

  // Load clinics and offers on component mount
  useEffect(() => {
    fetchClinics();
    fetchSpecialOffers();
  }, []);

  // Load clinic treatments when clinic changes
  useEffect(() => {
    if (watchClinicId) {
      fetchClinicTreatments(watchClinicId);
    } else {
      setTreatments([]);
    }
  }, [watchClinicId]);

  const fetchClinics = async () => {
    try {
      const response = await axios.get('/api/admin/clinics');
      setClinics(response.data);
    } catch (error) {
      console.error('Error fetching clinics:', error);
      toast.error('Failed to load clinics');
    }
  };

  const fetchClinicTreatments = async (clinicId) => {
    try {
      const response = await axios.get(`/api/admin/clinics/${clinicId}/treatments`);
      setTreatments(response.data);
    } catch (error) {
      console.error('Error fetching treatments:', error);
      toast.error('Failed to load treatments');
    }
  };

  const fetchSpecialOffers = async () => {
    try {
      setIsLoading(true);
      const response = await axios.get('/api/admin/special-offers');
      setSpecialOffers(response.data);
    } catch (error) {
      console.error('Error fetching special offers:', error);
      toast.error('Failed to load special offers');
    } finally {
      setIsLoading(false);
    }
  };

  const onSubmit = async (data) => {
    try {
      setIsLoading(true);
      
      if (isEditing) {
        await axios.put(`/api/admin/special-offers/${selectedOfferId}`, data);
        toast.success('Special offer updated successfully');
      } else {
        await axios.post('/api/admin/special-offers', data);
        toast.success('Special offer created successfully');
      }
      
      reset();
      setIsEditing(false);
      setSelectedOfferId(null);
      fetchSpecialOffers();
    } catch (error) {
      console.error('Error saving special offer:', error);
      toast.error(error.response?.data?.message || 'Failed to save special offer');
    } finally {
      setIsLoading(false);
    }
  };

  const handleEditOffer = (offer) => {
    setIsEditing(true);
    setSelectedOfferId(offer.id);
    
    // Populate form with offer data
    setValue('title', offer.title);
    setValue('description', offer.description);
    setValue('discountType', offer.discountType);
    setValue('discountValue', offer.discountValue);
    setValue('clinicId', offer.clinicId);
    setValue('treatmentIds', offer.treatmentIds || []);
    setValue('promoCode', offer.promoCode);
    setValue('expiryDate', new Date(offer.expiryDate).toISOString().split('T')[0]);
    setValue('isActive', offer.isActive);
    
    // Load treatments for this clinic
    fetchClinicTreatments(offer.clinicId);
  };

  const handleDeleteOffer = async (offerId) => {
    if (!confirm('Are you sure you want to delete this special offer?')) return;
    
    try {
      setIsLoading(true);
      await axios.delete(`/api/admin/special-offers/${offerId}`);
      toast.success('Special offer deleted');
      fetchSpecialOffers();
    } catch (error) {
      console.error('Error deleting special offer:', error);
      toast.error('Failed to delete special offer');
    } finally {
      setIsLoading(false);
    }
  };

  const generatePromoCode = () => {
    const selectedClinic = clinics.find(c => c.id === watchClinicId);
    const prefix = selectedClinic 
      ? selectedClinic.name.substring(0, 3).toUpperCase() 
      : 'OFF';
    
    const randomCode = Math.random().toString(36).substring(2, 7).toUpperCase();
    setValue('promoCode', `${prefix}-${randomCode}`);
  };

  return (
    <div className="container mx-auto p-4">
      <h1 className="text-2xl font-bold mb-6">
        {isEditing ? 'Edit Special Offer' : 'Create Special Offer'}
      </h1>
      
      <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
        {/* Form */}
        <div className="bg-white p-6 rounded-lg shadow">
          <form onSubmit={handleSubmit(onSubmit)}>
            <div className="mb-4">
              <label className="block text-sm font-medium mb-1">Clinic</label>
              <select 
                className="w-full p-2 border rounded"
                {...register('clinicId', { required: 'Please select a clinic' })}
                disabled={isLoading}
              >
                <option value="">Select a clinic</option>
                {clinics.map(clinic => (
                  <option key={clinic.id} value={clinic.id}>{clinic.name}</option>
                ))}
              </select>
              {errors.clinicId && <p className="text-red-500 text-xs mt-1">{errors.clinicId.message}</p>}
            </div>
            
            <div className="mb-4">
              <label className="block text-sm font-medium mb-1">Title</label>
              <input 
                type="text"
                className="w-full p-2 border rounded"
                placeholder="e.g., Summer Teeth Whitening Special"
                {...register('title', { required: 'Title is required' })}
                disabled={isLoading}
              />
              {errors.title && <p className="text-red-500 text-xs mt-1">{errors.title.message}</p>}
            </div>
            
            <div className="mb-4">
              <label className="block text-sm font-medium mb-1">Description</label>
              <textarea 
                className="w-full p-2 border rounded"
                rows="3"
                placeholder="Describe the special offer"
                {...register('description', { required: 'Description is required' })}
                disabled={isLoading}
              ></textarea>
              {errors.description && <p className="text-red-500 text-xs mt-1">{errors.description.message}</p>}
            </div>
            
            <div className="grid grid-cols-2 gap-4 mb-4">
              <div>
                <label className="block text-sm font-medium mb-1">Discount Type</label>
                <select 
                  className="w-full p-2 border rounded"
                  {...register('discountType')}
                  disabled={isLoading}
                >
                  <option value="PERCENTAGE">Percentage (%)</option>
                  <option value="FIXED_AMOUNT">Fixed Amount (£)</option>
                </select>
              </div>
              
              <div>
                <label className="block text-sm font-medium mb-1">Discount Value</label>
                <input 
                  type="number"
                  className="w-full p-2 border rounded"
                  step={watch('discountType') === 'PERCENTAGE' ? '1' : '0.01'}
                  min="0"
                  {...register('discountValue', { 
                    required: 'Discount value is required',
                    min: { value: 0, message: 'Must be a positive number' },
                    valueAsNumber: true
                  })}
                  disabled={isLoading}
                />
                {errors.discountValue && <p className="text-red-500 text-xs mt-1">{errors.discountValue.message}</p>}
              </div>
            </div>
            
            <div className="mb-4">
              <label className="block text-sm font-medium mb-1">Applicable Treatments</label>
              <div className="max-h-48 overflow-y-auto border rounded p-2">
                {treatments.length === 0 ? (
                  <p className="text-gray-500 text-sm">Select a clinic to see available treatments</p>
                ) : (
                  treatments.map(treatment => (
                    <div key={treatment.id} className="flex items-center mb-1">
                      <input 
                        type="checkbox"
                        value={treatment.id}
                        id={`treatment-${treatment.id}`}
                        className="mr-2"
                        {...register('treatmentIds', { required: 'Select at least one treatment' })}
                        disabled={isLoading}
                      />
                      <label htmlFor={`treatment-${treatment.id}`} className="text-sm">
                        {treatment.name} - £{treatment.price}
                      </label>
                    </div>
                  ))
                )}
              </div>
              {errors.treatmentIds && <p className="text-red-500 text-xs mt-1">{errors.treatmentIds.message}</p>}
            </div>
            
            <div className="grid grid-cols-2 gap-4 mb-4">
              <div>
                <label className="block text-sm font-medium mb-1">Promo Code</label>
                <div className="flex">
                  <input 
                    type="text"
                    className="w-full p-2 border rounded-l"
                    placeholder="e.g., SUMMER25"
                    {...register('promoCode', { required: 'Promo code is required' })}
                    disabled={isLoading}
                  />
                  <button 
                    type="button"
                    className="bg-gray-200 px-3 py-2 rounded-r"
                    onClick={generatePromoCode}
                    disabled={isLoading || !watchClinicId}
                  >
                    Generate
                  </button>
                </div>
                {errors.promoCode && <p className="text-red-500 text-xs mt-1">{errors.promoCode.message}</p>}
              </div>
              
              <div>
                <label className="block text-sm font-medium mb-1">Expiry Date</label>
                <input 
                  type="date"
                  className="w-full p-2 border rounded"
                  min={new Date().toISOString().split('T')[0]}
                  {...register('expiryDate', { required: 'Expiry date is required' })}
                  disabled={isLoading}
                />
                {errors.expiryDate && <p className="text-red-500 text-xs mt-1">{errors.expiryDate.message}</p>}
              </div>
            </div>
            
            <div className="mb-6">
              <label className="flex items-center">
                <input 
                  type="checkbox"
                  className="mr-2"
                  {...register('isActive')}
                  disabled={isLoading}
                />
                <span>Active</span>
              </label>
            </div>
            
            <div className="flex justify-end">
              {isEditing && (
                <button
                  type="button"
                  className="bg-gray-300 text-gray-800 px-4 py-2 rounded mr-2"
                  onClick={() => {
                    reset();
                    setIsEditing(false);
                    setSelectedOfferId(null);
                  }}
                  disabled={isLoading}
                >
                  Cancel
                </button>
              )}
              
              <button
                type="submit"
                className="bg-blue-600 text-white px-4 py-2 rounded"
                disabled={isLoading}
              >
                {isLoading ? 'Saving...' : isEditing ? 'Update Offer' : 'Create Offer'}
              </button>
            </div>
          </form>
        </div>
        
        {/* List of Offers */}
        <div>
          <h2 className="text-xl font-semibold mb-4">Special Offers</h2>
          
          {isLoading && specialOffers.length === 0 ? (
            <div className="text-center py-8">Loading...</div>
          ) : specialOffers.length === 0 ? (
            <p className="text-gray-500">No special offers found</p>
          ) : (
            <div className="space-y-4">
              {specialOffers.map(offer => {
                const clinic = clinics.find(c => c.id === offer.clinicId);
                
                return (
                  <div key={offer.id} className="bg-white p-4 rounded-lg shadow">
                    <div className="flex justify-between">
                      <h3 className="font-bold">{offer.title}</h3>
                      <span className={`text-xs px-2 py-1 rounded-full ${offer.isActive ? 'bg-green-100 text-green-800' : 'bg-gray-100 text-gray-800'}`}>
                        {offer.isActive ? 'Active' : 'Inactive'}
                      </span>
                    </div>
                    
                    <p className="text-sm text-gray-600 mt-1">{offer.description.substring(0, 100)}...</p>
                    
                    <div className="mt-2 grid grid-cols-2 gap-x-4 gap-y-1 text-sm">
                      <div><span className="font-medium">Clinic:</span> {clinic?.name || 'Unknown'}</div>
                      <div><span className="font-medium">Promo Code:</span> {offer.promoCode}</div>
                      
                      <div>
                        <span className="font-medium">Discount:</span> 
                        {offer.discountType === 'PERCENTAGE' 
                          ? `${offer.discountValue}%` 
                          : `£${offer.discountValue}`}
                      </div>
                      
                      <div>
                        <span className="font-medium">Expires:</span> 
                        {new Date(offer.expiryDate).toLocaleDateString()}
                      </div>
                    </div>
                    
                    <div className="mt-3 flex justify-end space-x-2">
                      <button
                        onClick={() => handleEditOffer(offer)}
                        className="text-blue-600 hover:text-blue-800 text-sm"
                        disabled={isLoading}
                      >
                        Edit
                      </button>
                      <button
                        onClick={() => handleDeleteOffer(offer.id)}
                        className="text-red-600 hover:text-red-800 text-sm"
                        disabled={isLoading}
                      >
                        Delete
                      </button>
                    </div>
                  </div>
                );
              })}
            </div>
          )}
        </div>
      </div>
    </div>
  );
}

import React, { useState, useEffect } from 'react';
import { useForm, useFieldArray } from 'react-hook-form';
import axios from 'axios';
import toast from 'react-hot-toast';

export default function TreatmentPackageManager() {
  const [clinics, setClinics] = useState([]);
  const [treatments, setTreatments] = useState([]);
  const [packages, setPackages] = useState([]);
  const [isEditing, setIsEditing] = useState(false);
  const [selectedPackageId, setSelectedPackageId] = useState(null);
  const [isLoading, setIsLoading] = useState(false);
  const [calculatedOriginalPrice, setCalculatedOriginalPrice] = useState(0);

  const { register, control, handleSubmit, reset, setValue, watch, formState: { errors } } = useForm({
    defaultValues: {
      title: '',
      description: '',
      clinicId: '',
      includedTreatments: [{ treatmentId: '', treatmentName: '', treatmentPrice: 0, quantity: 1 }],
      includedServices: [{ serviceType: 'TRANSFER', name: 'Airport Transfer', description: '', quantity: 1 }],
      totalPrice: 0,
      promoCode: '',
      expiryDate: new Date().toISOString().split('T')[0],
      isActive: true
    }
  });

  // Field arrays for dynamic fields
  const { 
    fields: treatmentFields, 
    append: appendTreatment, 
    remove: removeTreatment 
  } = useFieldArray({ control, name: 'includedTreatments' });
  
  const { 
    fields: serviceFields, 
    append: appendService, 
    remove: removeService 
  } = useFieldArray({ control, name: 'includedServices' });

  const watchClinicId = watch('clinicId');
  const watchIncludedTreatments = watch('includedTreatments');

  // Load initial data
  useEffect(() => {
    fetchClinics();
    fetchPackages();
  }, []);

  // Load clinic treatments when clinic changes
  useEffect(() => {
    if (watchClinicId) {
      fetchClinicTreatments(watchClinicId);
    } else {
      setTreatments([]);
    }
  }, [watchClinicId]);

  // Calculate original price when treatments change
  useEffect(() => {
    if (watchIncludedTreatments) {
      let total = 0;
      
      watchIncludedTreatments.forEach(item => {
        if (item.treatmentId && item.quantity) {
          const treatment = treatments.find(t => t.id === item.treatmentId);
          if (treatment) {
            total += treatment.price * item.quantity;
          } else if (item.treatmentPrice) {
            // For editing existing packages that have stored prices
            total += item.treatmentPrice * item.quantity;
          }
        }
      });
      
      setCalculatedOriginalPrice(total);
    }
  }, [watchIncludedTreatments, treatments]);

  const fetchClinics = async () => {
    try {
      const response = await axios.get('/api/admin/clinics');
      setClinics(response.data);
    } catch (error) {
      console.error('Error fetching clinics:', error);
      toast.error('Failed to load clinics');
    }
  };

  const fetchClinicTreatments = async (clinicId) => {
    try {
      const response = await axios.get(`/api/admin/clinics/${clinicId}/treatments`);
      setTreatments(response.data);
    } catch (error) {
      console.error('Error fetching treatments:', error);
      toast.error('Failed to load treatments');
    }
  };

  const fetchPackages = async () => {
    try {
      setIsLoading(true);
      const response = await axios.get('/api/admin/treatment-packages');
      setPackages(response.data);
    } catch (error) {
      console.error('Error fetching packages:', error);
      toast.error('Failed to load treatment packages');
    } finally {
      setIsLoading(false);
    }
  };

  const onSubmit = async (data) => {
    try {
      setIsLoading(true);
      
      // Prepare data - add treatment details for each included treatment
      const preparedData = {
        ...data,
        originalPrice: calculatedOriginalPrice,
        includedTreatments: data.includedTreatments.map(item => {
          const treatment = treatments.find(t => t.id === item.treatmentId);
          return {
            ...item,
            treatmentName: treatment?.name || item.treatmentName,
            treatmentPrice: treatment?.price || item.treatmentPrice
          };
        })
      };
      
      if (isEditing) {
        await axios.put(`/api/admin/treatment-packages/${selectedPackageId}`, preparedData);
        toast.success('Treatment package updated successfully');
      } else {
        await axios.post('/api/admin/treatment-packages', preparedData);
        toast.success('Treatment package created successfully');
      }
      
      reset();
      setIsEditing(false);
      setSelectedPackageId(null);
      fetchPackages();
    } catch (error) {
      console.error('Error saving package:', error);
      toast.error(error.response?.data?.message || 'Failed to save treatment package');
    } finally {
      setIsLoading(false);
    }
  };

  const handleEditPackage = (pkg) => {
    setIsEditing(true);
    setSelectedPackageId(pkg.id);
    
    // Populate form with package data
    setValue('title', pkg.title);
    setValue('description', pkg.description);
    setValue('clinicId', pkg.clinicId);
    setValue('includedTreatments', pkg.includedTreatments);
    setValue('includedServices', pkg.includedServices);
    setValue('totalPrice', pkg.totalPrice);
    setValue('promoCode', pkg.promoCode);
    setValue('expiryDate', new Date(pkg.expiryDate).toISOString().split('T')[0]);
    setValue('isActive', pkg.isActive);
    
    // Load treatments for this clinic
    fetchClinicTreatments(pkg.clinicId);
  };

  const handleDeletePackage = async (packageId) => {
    if (!confirm('Are you sure you want to delete this treatment package?')) return;
    
    try {
      setIsLoading(true);
      await axios.delete(`/api/admin/treatment-packages/${packageId}`);
      toast.success('Treatment package deleted');
      fetchPackages();
    } catch (error) {
      console.error('Error deleting package:', error);
      toast.error('Failed to delete treatment package');
    } finally {
      setIsLoading(false);
    }
  };

  const handleTreatmentChange = (index, treatmentId) => {
    const treatment = treatments.find(t => t.id === treatmentId);
    if (treatment) {
      setValue(`includedTreatments.${index}.treatmentName`, treatment.name);
      setValue(`includedTreatments.${index}.treatmentPrice`, treatment.price);
    }
  };

  const generatePromoCode = () => {
    const selectedClinic = clinics.find(c => c.id === watchClinicId);
    const prefix = selectedClinic 
      ? selectedClinic.name.substring(0, 3).toUpperCase() 
      : 'PKG';
    
    const randomCode = Math.random().toString(36).substring(2, 7).toUpperCase();
    setValue('promoCode', `${prefix}-${randomCode}`);
  };

  return (
    <div className="container mx-auto p-4">
      <h1 className="text-2xl font-bold mb-6">
        {isEditing ? 'Edit Treatment Package' : 'Create Treatment Package'}
      </h1>
      
      <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
        {/* Form */}
        <div className="bg-white p-6 rounded-lg shadow">
          <form onSubmit={handleSubmit(onSubmit)}>
            <div className="mb-4">
              <label className="block text-sm font-medium mb-1">Clinic</label>
              <select 
                className="w-full p-2 border rounded"
                {...register('clinicId', { required: 'Please select a clinic' })}
                disabled={isLoading}
              >
                <option value="">Select a clinic</option>
                {clinics.map(clinic => (
                  <option key={clinic.id} value={clinic.id}>{clinic.name}</option>
                ))}
              </select>
              {errors.clinicId && <p className="text-red-500 text-xs mt-1">{errors.clinicId.message}</p>}
            </div>
            
            <div className="mb-4">
              <label className="block text-sm font-medium mb-1">Package Title</label>
              <input 
                type="text"
                className="w-full p-2 border rounded"
                placeholder="e.g., Complete Smile Makeover Package"

// contexts/QuoteContext.jsx
import React, { createContext, useContext, useState, useEffect } from 'react';
import axios from 'axios';

const QuoteContext = createContext();

export function QuoteProvider({ children }) {
  const [selectedTreatments, setSelectedTreatments] = useState([]);
  const [selectedClinic, setSelectedClinic] = useState(null);
  const [promoCode, setPromoCode] = useState(null);
  const [promoDetails, setPromoDetails] = useState(null);
  const [isValidatingPromo, setIsValidatingPromo] = useState(false);
  const [promoError, setPromoError] = useState(null);
  
  // Calculate original price (without discounts)
  const originalTotal = selectedTreatments.reduce(
    (sum, treatment) => sum + treatment.price * (treatment.quantity || 1),
    0
  );
  
  // Calculate discounted price
  const [discountedTotal, setDiscountedTotal] = useState(originalTotal);
  const [discountAmount, setDiscountAmount] = useState(0);
  
  // Validate promo code when it changes
  useEffect(() => {
    async function validatePromoCode() {
      if (!promoCode) {
        setPromoDetails(null);
        setPromoError(null);
        return;
      }
      
      setIsValidatingPromo(true);
      setPromoError(null);
      
      try {
        const response = await axios.post('/api/promo-codes/validate', {
          promoCode
        });
        
        setPromoDetails(response.data);
        
        // If it's a package, we might want to pre-select the clinic
        if (response.data.type === 'TREATMENT_PACKAGE') {
          setSelectedClinic(response.data.package.clinic);
        } else if (response.data.type === 'SPECIAL_OFFER') {
          // For special offers, we might also want to pre-select the clinic
          setSelectedClinic(response.data.offer.clinic);
        }
      } catch (error) {
        console.error('Error validating promo code:', error);
        setPromoDetails(null);
        setPromoError(error.response?.data?.message || 'Invalid promo code');
      } finally {
        setIsValidatingPromo(false);
      }
    }
    
    if (promoCode) {
      validatePromoCode();
    } else {
      setPromoDetails(null);
    }
  }, [promoCode]);
  
  // Calculate final price whenever treatments or promo changes
  useEffect(() => {
    if (!promoDetails || selectedTreatments.length === 0) {
      setDiscountedTotal(originalTotal);
      setDiscountAmount(0);
      return;
    }
    
    if (promoDetails.type === 'SPECIAL_OFFER') {
      const offer = promoDetails.offer;
      
      // Check if any selected treatments are eligible for the discount
      const hasEligibleTreatment = selectedTreatments.some(treatment => 
        offer.treatmentIds.includes(treatment.id)
      );
      
      if (hasEligibleTreatment) {
        let discount = 0;
        
        if (offer.discountType === 'PERCENTAGE') {
          discount = (originalTotal * offer.discountValue) / 100;
        } else if (offer.discountType === 'FIXED_AMOUNT') {
          discount = Math.min(originalTotal, offer.discountValue);
        }
        
        setDiscountAmount(discount);
        setDiscountedTotal(originalTotal - discount);
      } else {
        setDiscountedTotal(originalTotal);
        setDiscountAmount(0);
      }
    } else if (promoDetails.type === 'TREATMENT_PACKAGE') {
      // For packages, we use the package price directly
      const pkg = promoDetails.package;
      
      // Check if selected treatments match the package
      const packageTreatmentIds = pkg.includedTreatments.map(item => item.treatmentId);
      const selectedTreatmentIds = selectedTreatments.map(item => item.id);
      
      // Simple check - might need more sophisticated logic in real app
      const hasAllPackageTreatments = packageTreatmentIds.every(id => 
        selectedTreatmentIds.includes(id)
      );
      
      if (hasAllPackageTreatments && selectedClinic?.id === pkg.clinicId) {
        setDiscountedTotal(pkg.totalPrice);
        setDiscountAmount(originalTotal - pkg.totalPrice);
      } else {
        setDiscountedTotal(originalTotal);
        setDiscountAmount(0);
      }
    }
  }, [selectedTreatments, promoDetails, originalTotal, selectedClinic]);
  
  const clearPromoCode = () => {
    setPromoCode(null);
    setPromoDetails(null);
    setPromoError(null);
  };
  
  const value = {
    // Treatments
    selectedTreatments,
    setSelectedTreatments,
    
    // Clinic
    selectedClinic,
    setSelectedClinic,
    
    // Promo code
    promoCode,
    setPromoCode,
    clearPromoCode,
    promoDetails,
    isValidatingPromo,
    promoError,
    
    // Pricing
    originalTotal,
    discountedTotal,
    discountAmount,
    
    // Helper function to apply a treatment package
    applyTreatmentPackage: (packageData) => {
      // Set the promo code
      setPromoCode(packageData.promoCode);
      
      // Optionally pre-select treatments
      // This depends on your UX flow
    }
  };
  
  return (
    <QuoteContext.Provider value={value}>
      {children}
    </QuoteContext.Provider>
  );
}

export function useQuoteContext() {
  return useContext(QuoteContext);
}
