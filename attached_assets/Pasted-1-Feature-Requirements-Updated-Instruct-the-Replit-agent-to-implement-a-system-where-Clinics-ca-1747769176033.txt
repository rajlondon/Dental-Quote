// Promotion status reducer with scheduling and homepage features
const promotionReducer = (state, action) => {
  switch (action.type) {
    case 'SUBMIT_FOR_APPROVAL':
      return { ...state, status: 'PENDING_APPROVAL', submitted_at: new Date().toISOString() };
    case 'APPROVE':
      return { 
        ...state, 
        status: 'APPROVED', 
        reviewed_by: action.payload.adminId,
        reviewed_at: new Date().toISOString(),
        approval_notes: action.payload.notes,
        // If admin modified dates during approval
        ...(action.payload.modifiedDates ? {
          start_date: action.payload.startDate,
          end_date: action.payload.endDate,
          admin_modified_dates: true,
          original_start_date: state.start_date,
          original_end_date: state.end_date
        } : {})
      };
    case 'UPDATE_HOMEPAGE_DISPLAY':
      return {
        ...state,
        display_on_homepage: action.payload.displayOnHomepage,
        homepage_priority: action.payload.priority,
        homepage_image_url: action.payload.imageUrl,
        homepage_short_description: action.payload.shortDescription
      };
    // Additional cases...
    default:
      return state;
  }
};

// Example: Admin approval with scheduling
async function approvePromotionWithScheduling(
  promotionId: string, 
  notes: string,
  startDate: string,
  endDate: string,
  displayOnHomepage: boolean,
  homepagePriority: number = 5
) {
  try {
    const response = await fetch(`/api/admin/promotions/${promotionId}/approve`, {
      method: 'PUT',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({ 
        notes,
        startDate,
        endDate,
        displayOnHomepage,
        homepagePriority
      })
    });
    
    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.message);
    }
    
    const result = await response.json();
    toast.success('Promotion approved and scheduled successfully');
    return result;
  } catch (error) {
    toast.error(`Failed to approve promotion: ${error.message}`);
    throw error;
  }
}

// Example: Get featured promotions for homepage
async function getFeaturedPromotions() {
  try {
    const response = await fetch('/api/promotions/featured');
    
    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.message);
    }
    
    return await response.json();
  } catch (error) {
    console.error('Failed to fetch featured promotions:', error);
    return [];
  }
}

// Example: Calculate end date based on duration
function calculateEndDate(startDate: Date, durationInDays: number): Date {
  const endDate = new Date(startDate);
  endDate.setDate(startDate.getDate() + durationInDays);
  return endDate;
}
