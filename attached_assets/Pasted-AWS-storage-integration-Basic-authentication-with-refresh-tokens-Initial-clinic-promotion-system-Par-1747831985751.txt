// client/src/pages/patient/PaymentsPage.tsx

function PaymentsPage() {
  const { data: payments, isLoading } = useQuery(
    ["patient-payments", currentUser.id],
    () => apiClient.get(`/api/patients/${currentUser.id}/payments`)
  );
  
  // Process payment via Stripe
  async function processStripePayment(paymentId) {
    const payment = payments.find(p => p.id === paymentId);
    
    // Calculate amount with Stripe fee
    const amountWithFee = payment.amount * 1.029 + 0.30; // 2.9% + $0.30 Stripe fee
    
    // Create Stripe payment intent
    const { clientSecret } = await apiClient.post(`/api/payments/stripe/create-payment-intent`, {
      paymentId,
      amountWithFee
    });
    
    // Open Stripe payment element
    const stripe = await loadStripe(process.env.STRIPE_PUBLIC_KEY);
    const result = await stripe.confirmCardPayment(clientSecret, {
      payment_method: {
        card: elements.getElement(CardElement),
        billing_details: {
          name: currentUser.name,
          email: currentUser.email
        }
      }
    });
    
    if (result.error) {
      toast.error(`Payment failed: ${result.error.message}`);
    } else {
      // Update payment status
      await apiClient.put(`/api/payments/${paymentId}/complete`, {
        transactionId: result.paymentIntent.id,
        method: "STRIPE"
      });
      
      // Notify clinic
      await apiClient.post(`/api/notifications`, {
        recipientId: payment.clinicId,
        type: "PAYMENT_RECEIVED",
        paymentId,
        patientId: currentUser.id,
        amount: payment.amount,
        paymentType: payment.type
      });
      
      toast.success("Payment processed successfully!");
    }
  }
  
  // Process bank transfer
  async function initiateWireTransfer(paymentId) {
    const payment = payments.find(p => p.id === paymentId);
    
    // Get bank details
    const { bankDetails } = await apiClient.get(`/api/clinics/${payment.clinicId}/bank-details`);
    
    // Mark payment as pending bank transfer
    await apiClient.put(`/api/payments/${paymentId}/initiate-transfer`);
    
    // Show bank details to patient
    setBankDetails(bankDetails);
    setShowBankDetailsModal(true);
    
    // Notify clinic
    await apiClient.post(`/api/notifications`, {
      recipientId: payment.clinicId,
      type: "BANK_TRANSFER_INITIATED",
      paymentId,
      patientId: currentUser.id
    });
  }
}

// client/src/components/messaging/MessageCenter.tsx

function MessageCenter({ portalType, userId }) {
  const [conversations, setConversations] = useState([]);
  const [activeConversation, setActiveConversation] = useState(null);
  const [message, setMessage] = useState("");
  
  // Load conversations based on portal type
  useEffect(() => {
    async function loadConversations() {
      let endpoint;
      
      switch (portalType) {
        case "PATIENT":
          endpoint = `/api/messages/patient/${userId}/conversations`;
          break;
        case "CLINIC":
          endpoint = `/api/messages/clinic/${userId}/conversations`;
          break;
        case "ADMIN":
          endpoint = `/api/messages/admin/conversations`;
          break;
      }
      
      const response = await apiClient.get(endpoint);
      setConversations(response.data);
    }
    
    loadConversations();
  }, [portalType, userId]);
  
  // Load messages for active conversation
  useEffect(() => {
    if (!activeConversation) return;
    
    async function loadMessages() {
      const response = await apiClient.get(`/api/messages/conversations/${activeConversation.id}/messages`);
      setActiveConversation(prev => ({
        ...prev,
        messages: response.data
      }));
      
      // Mark messages as read
      await apiClient.put(`/api/messages/conversations/${activeConversation.id}/read`);
    }
    
    loadMessages();
    
    // Set up real-time updates
    const socket = io(process.env.SOCKET_URL);
    socket.on(`new-message-${activeConversation.id}`, newMessage => {
      setActiveConversation(prev => ({
        ...prev,
        messages: [...prev.messages, newMessage]
      }));
    });
    
    return () => {
      socket.disconnect();
    };
  }, [activeConversation?.id]);
  
  // Send message
  async function sendMessage() {
    if (!message.trim()) return;
    
    await apiClient.post(`/api/messages/conversations/${activeConversation.id}/messages`, {
      content: message,
      senderId: userId,
      senderType: portalType
    });
    
    setMessage("");
  }
  
  // Start new conversation
  async function startNewConversation(recipientId, recipientType) {
    const response = await apiClient.post(`/api/messages/conversations`, {
      participants: [
        { id: userId, type: portalType },
        { id: recipientId, type: recipientType }
      ],
      title: `Conversation with ${recipientType.toLowerCase()}`
    });
    
    setConversations([...conversations, response.data]);
    setActiveConversation(response.data);
  }
}
Implementation Strategy
To ensure complete functionality across all portals, I recommend implementing these features in the following order:

Phase 1: Core Data Flow
Messaging System - As you mentioned, this is a critical component
Patient Portal Basic Functionality - Documents, appointments, treatment plans
Clinic Portal Basic Functionality - Treatment management, appointment handling
Phase 2: Financial & Travel Features
Payment Processing - Implement deposit and full payment functionality
Hotel & Flights Section - Reimplement the travel section
Special Offers Integration - Complete integration with appointments
Phase 3: Advanced Features
Dental Chart - Interactive dental chart with treatment history
Support System - Ticket management and resolution tracking
Treatment Packages - Advanced package creation and management
Testing Strategy
For each implemented feature, we should:

Unit Test - Test individual components in isolation
Integration Test - Test data flow between components
End-to-End Test - Test complete user workflows
