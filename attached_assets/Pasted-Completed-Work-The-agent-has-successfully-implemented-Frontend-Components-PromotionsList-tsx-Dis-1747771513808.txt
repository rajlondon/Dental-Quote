// Create query hooks for promotions
export function usePromotions(clinicId: string) {
  return useQuery(['promotions', clinicId], 
    () => fetch(`/api/clinic/promotions?clinicId=${clinicId}`).then(res => res.json())
  );
}

export function usePendingApprovals() {
  return useQuery('pendingApprovals', 
    () => fetch('/api/admin/promotions/pending').then(res => res.json())
  );
}

export function useFeaturedPromotions() {
  return useQuery('featuredPromotions', 
    () => fetch('/api/promotions/featured').then(res => res.json())
  );
}
