import React from 'react';
import { Router as WouterRouter, Route, Switch } from 'wouter';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { GlobalAuthProvider } from '@/contexts/GlobalAuthProvider';
import { Toaster } from '@/components/ui/toaster';
import ProtectedRoute from '@/lib/protected-route';

// Main Pages
import Home from '@/pages/Home';
import PortalLoginPage from '@/pages/PortalLoginPage';
import YourQuotePage2 from '@/pages/YourQuotePage2';
import MatchedClinicsPage from '@/pages/MatchedClinicsPage';
import QuoteRequestPage from '@/pages/QuoteRequestPage';
import QuoteResultsPage from '@/pages/QuoteResultsPage';
import PaymentConfirmationPage from '@/pages/PaymentConfirmationPage';
import BookingPage from '@/pages/BookingPage';
import DepositPaymentPage from '@/pages/DepositPaymentPage';
import FinalPaymentPage from '@/pages/FinalPaymentPage';

// Portal Pages
import PatientPortalPage from '@/pages/PatientPortalPage';
import ClinicPortalPage from '@/pages/ClinicPortalPage';
import AdminPortalPage from '@/pages/AdminPortalPage';
import StableClinicPortal from '@/components/StableClinicPortal';

// Testing and Support Pages
import PortalTestingHub from '@/pages/PortalTestingHub';
import PortalCommunicationTester from '@/pages/PortalCommunicationTester';

// Blog and Information Pages
import BlogPage from '@/pages/BlogPage';
import DentalImplantsPage from '@/pages/DentalImplantsPage';
import VeneersPage from '@/pages/VeneersPage';
import HollywoodSmilePage from '@/pages/HollywoodSmilePage';
import FullMouthPage from '@/pages/FullMouthPage';
import FAQPage from '@/pages/FAQPage';
import TeamPage from '@/pages/TeamPage';
import HowItWorks from '@/pages/HowItWorks';

// Auth Pages
import EmailVerifiedPage from '@/pages/EmailVerifiedPage';
import VerifyEmailPage from '@/pages/VerifyEmailPage';
import ForgotPasswordPage from '@/pages/ForgotPasswordPage';
import ResetPasswordPage from '@/pages/ResetPasswordPage';

// Error Pages
import ErrorPage from '@/pages/ErrorPage';

const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      retry: 2,
      refetchOnWindowFocus: false,
      staleTime: 5 * 60 * 1000, // 5 minutes
    },
  },
});

export default function AppRouter() {
  return (
    <QueryClientProvider client={queryClient}>
      <GlobalAuthProvider>
        <WouterRouter>
          <Switch>
            {/* Homepage and Main Routes */}
            <Route path="/" component={Home} />
            <Route path="/home" component={Home} />
            
            {/* Quote and Booking Flow */}
            <Route path="/your-quote" component={YourQuotePage2} />
            <Route path="/quote-request" component={QuoteRequestPage} />
            <Route path="/quote-results" component={QuoteResultsPage} />
            <Route path="/matched-clinics" component={MatchedClinicsPage} />
            <Route path="/payment-confirmation" component={PaymentConfirmationPage} />
            <Route path="/booking" component={BookingPage} />
            <Route path="/deposit-payment" component={DepositPaymentPage} />
            <Route path="/final-payment" component={FinalPaymentPage} />

            {/* Authentication */}
            <Route path="/portal-login" component={PortalLoginPage} />
            <Route path="/email-verified" component={EmailVerifiedPage} />
            <Route path="/verify-email" component={VerifyEmailPage} />
            <Route path="/forgot-password" component={ForgotPasswordPage} />
            <Route path="/reset-password" component={ResetPasswordPage} />

            {/* Protected Portal Routes */}
            <ProtectedRoute 
              path="/patient-portal" 
              component={PatientPortalPage} 
              requiredRole="patient" 
            />
            <ProtectedRoute 
              path="/client-portal" 
              component={PatientPortalPage} 
              requiredRole="patient" 
            />
            <ProtectedRoute 
              path="/clinic-portal" 
              component={StableClinicPortal} 
              requiredRole="clinic_staff" 
            />
            <ProtectedRoute 
              path="/admin-portal" 
              component={AdminPortalPage} 
              requiredRole="admin" 
            />

            {/* Information Pages */}
            <Route path="/blog" component={BlogPage} />
            <Route path="/dental-implants" component={DentalImplantsPage} />
            <Route path="/veneers" component={VeneersPage} />
            <Route path="/hollywood-smile" component={HollywoodSmilePage} />
            <Route path="/full-mouth-reconstruction" component={FullMouthPage} />
            <Route path="/faq" component={FAQPage} />
            <Route path="/team" component={TeamPage} />
            <Route path="/how-it-works" component={HowItWorks} />

            {/* Testing and Development */}
            <Route path="/portal-testing" component={PortalTestingHub} />
            <Route path="/communication-test" component={PortalCommunicationTester} />

            {/* Error Handling */}
            <Route path="/error" component={ErrorPage} />
            
            {/* Catch-all 404 */}
            <Route>
              {() => (
                <div className="min-h-screen bg-gray-50 flex items-center justify-center">
                  <div className="text-center">
                    <h1 className="text-4xl font-bold text-gray-900 mb-4">404 - Page Not Found</h1>
                    <p className="text-gray-600 mb-8">The page you're looking for doesn't exist.</p>
                    <a 
                      href="/" 
                      className="bg-blue-600 text-white px-6 py-3 rounded-md hover:bg-blue-700 transition-colors"
                    >
                      Return Home
                    </a>
                  </div>
                </div>
              )}
            </Route>
          </Switch>
          <Toaster />
        </WouterRouter>
      </GlobalAuthProvider>
    </QueryClientProvider>
  );
}
